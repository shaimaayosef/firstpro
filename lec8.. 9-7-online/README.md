var myName ='shaimaa'
var jop = 'developer'
var str = `my name is ${myName} and i am a ${jop}`
console.log(str) //my name is shaimaa and i am a developer
console.log(jop.length); //9
console.log(jop.charAt(0)); //d
console.log(jop.indexOf('v')); //2

    name:'yellowFlower',
    price: 10,
    category:'flower'
}
console.log(product); // {name:'yellowFlower',price: 10,category:'flower'}
console.log(product.name);   //yellowFlower
console.log(product.price);  //10
